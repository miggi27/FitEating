import cv2
import torch
import numpy as np
from ultralytics import YOLO
from sam2.build_sam import build_sam2
from sam2.sam2_image_predictor import SAM2ImagePredictor

# 1. 모델 로드
# YOLO11 (음식 탐지용 - 박스 모델도 가능)
yolo_model = YOLO("yolo11n.pt") 

# SAM 2 로드 (설정 파일과 체크포인트 필요)
sam2_checkpoint = "sam2_hiera_large.pt"
model_cfg = "sam2_hiera_l.yaml"
sam2_model = build_sam2(model_cfg, sam2_checkpoint, device="cuda" if torch.cuda.is_available() else "cpu")
predictor = SAM2ImagePredictor(sam2_model)

# 2. 이미지 읽기
img_path = "food_image.jpg"
image = cv2.imread(img_path)
image_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

# 3. YOLO11로 음식 박스 추출
results = yolo_model.predict(image, conf=0.25)
boxes = results[0].boxes.xyxy.cpu().numpy() # [xmin, ymin, xmax, ymax]

# 4. SAM 2로 정교한 세그먼트 생성
predictor.set_image(image_rgb)

if len(boxes) > 0:
    # YOLO가 찾은 모든 박스를 SAM에게 입력값으로 전달
    masks, scores, _ = predictor.predict(
        box=boxes,
        multimask_output=False
    )

    # 5. 결과 시각화 및 마스크 적용
    for i, mask in enumerate(masks):
        # mask는 불리언 배열 (True/False)
        color = np.random.randint(0, 255, (3,)).tolist()
        image[mask[0]] = image[mask[0]] * 0.5 + np.array(color) * 0.5

# 결과 저장
cv2.imwrite("sam3-result-test.jpg", image)
print("분석 완료! sam3-result-test.jpg를 확인하세요.")
